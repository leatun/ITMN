`include "_parameter.v"

module Unified_PE
(
    input clk,
    input reset,
    
    // --- Tin hieu dieu khien ---
    input [1:0] op_mode,   // Chon che do: MAC, MUL, hay ADD
    input clear_acc,       // Xoa thanh ghi tich luy (dung khi bat dau chuoi MAC moi)

    // --- Du lieu dau vao (Signed 16-bit) ---
    input signed [`DATA_WIDTH-1:0] in_A,
    input signed [`DATA_WIDTH-1:0] in_B,

    // --- Du lieu dau ra (Signed 16-bit) ---
    output reg signed [`DATA_WIDTH-1:0] out_val
);

    // --- Hang so gioi han cho Saturation (chong tran) ---
    localparam signed [`DATA_WIDTH-1:0] MAX_POS = 16'sh7FFF; // +32767
    localparam signed [`DATA_WIDTH-1:0] MIN_NEG = 16'sh8000; // -32768

    // --- Bien noi bo ---
    
    // 1. Ket qua nhan tho (32-bit)
    wire signed [2*`DATA_WIDTH-1:0] mult_raw;
    
    // 2. Ket qua nhan da dich bit (dieu chinh fixed-point)
    // Dich phai 12 bit de quay ve ti le Q3.12
    wire signed [2*`DATA_WIDTH-1:0] mult_shifted;

    // 3. Ket qua tinh toan tam thoi (truoc khi bao hoa)
    // Can 32 bit de chua an toan cac phep cong don ma khong bi tran ngay lap tuc
    reg signed [31:0] temp_result;

    // 4. Thanh ghi tich luy (Accumulator Register)
    // Luu trang thai hien tai cua PE
    reg signed [`DATA_WIDTH-1:0] acc_reg;


    // --- LOGIC TINH TOAN (Combinational) ---

    // Buoc A: Nhan (luon thuc hien, nhung co the khong dung neu o mode ADD)
    assign mult_raw = in_A * in_B;
    
    // Buoc B: Dich bit so hoc (giu nguyen dau)
    assign mult_shifted = mult_raw >>> `FRAC_BITS;


    // Buoc C: MUX chon phep toan dua tren op_mode
    always @(*) begin
        case (op_mode)
            `MODE_MAC: begin 
                // MAC: Cong ket qua nhan moi vao gia tri tich luy cu
                // Out = Acc + (A * B)
                temp_result = acc_reg + mult_shifted;
            end
            
            `MODE_MUL: begin
                // MUL: Chi lay ket qua nhan
                // Out = (A * B)
                temp_result = mult_shifted; 
            end
            
            `MODE_ADD: begin
                // ADD: Cong hai dau vao
                // Out = A + B
                // (Khong can dich bit vi phep cong cung scale)
                temp_result = in_A + in_B;
            end
            
            default: temp_result = 0;
        endcase
    end


    // --- LOGIC THANH GHI VA BAO HOA (Sequential) ---
    
    always @(posedge clk or posedge reset) begin
        if (reset) begin
            acc_reg <= 0;
            out_val <= 0;
        end else if (clear_acc) begin
            // Lenh xoa tich luy (thuong dung truoc khi bat dau tinh dot product moi)
            acc_reg <= 0;
            out_val <= 0;
        end else begin
            // --- Logic bao hoa (Saturation / Clamping) ---
            // Kiem tra xem temp_result (32-bit) co vuot qua gioi han 16-bit hay khong
            
            if (temp_result > 32767) begin
                acc_reg <= MAX_POS;
            end else if (temp_result < -32768) begin
                acc_reg <= MIN_NEG;
            end else begin
                // Neu nam trong gioi han, cat lay 16 bit thap
                acc_reg <= temp_result[`DATA_WIDTH-1:0];
            end
            
            // Cap nhat dau ra (dau ra chinh la gia tri trong thanh ghi tich luy)
            out_val <= (temp_result > 32767) ? MAX_POS : 
                       ((temp_result < -32768) ? MIN_NEG : temp_result[`DATA_WIDTH-1:0]);
        end
    end

endmodule
