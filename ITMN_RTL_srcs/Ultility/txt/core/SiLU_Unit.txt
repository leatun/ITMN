`include "_parameter.v"

module SiLU_Unit
(
    input clk,
    input signed [`DATA_WIDTH-1:0] in_data,
    output reg signed [`DATA_WIDTH-1:0] out_data
);

    // 65536-entry ROM for all 16-bit patterns
    reg signed [`DATA_WIDTH-1:0] rom [0:(1<<`DATA_WIDTH)-1];

    // Address is raw bit pattern of input
    wire [`DATA_WIDTH-1:0] addr = in_data;

    initial begin
        $readmemh("silu_rom.mem", rom);
    end

    always @(posedge clk) begin
        out_data <= rom[addr];
    end

endmodule
