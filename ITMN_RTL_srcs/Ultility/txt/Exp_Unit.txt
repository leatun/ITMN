`include "_parameter.v"

module Exp_Unit
(
    input clk,
    input signed [15:0] in_data, 
    output reg signed [15:0] out_data
);

    (* rom_style = "block" *) reg [15:0] rom [0:65535];

    initial begin
        // Doc file py
        $readmemh("exp_rom.mem", rom);
    end

    wire [15:0] addr;
    assign addr = in_data; 

    always @(posedge clk) begin
        out_data <= rom[addr];
    end

endmodule
